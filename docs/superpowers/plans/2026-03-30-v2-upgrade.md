```python
"""
Shared helpers for LexLink MCP tools.

Extracts common patterns from server.py to reduce duplication.
All helpers preserve exact existing behavior.
"""

import functools
import inspect
import logging
import os
from typing import Optional, Union

from mcp.types import ToolAnnotations

from .errors import ErrorCode, create_error_response

logger = logging.getLogger(__name__)


# Shared tool annotations — all LexLink tools are read-only, non-destructive, idempotent
TOOL_ANNOTATIONS = ToolAnnotations(
    readOnlyHint=True,
    destructiveHint=False,
    idempotentHint=True,
)


def stringify_id(*values: Optional[Union[str, int]]) -> tuple:
    """
    Convert id/mst/jo values to strings if they are integers.

    LLMs may extract numbers as ints from XML, but API params expect strings.

    Args:
        *values: Optional values that may be str, int, or None

    Returns:
        Tuple of converted values (str or None)

    Examples:
        >>> stringify_id(12345, None, 300)
        ('12345', None, '300')
    """
    return tuple(str(v) if v is not None else None for v in values)


class ToolValidationError(ValueError):
    """ValueError subclass that carries hints for the error response."""

    def __init__(self, message: str, hints: Optional[list[str]] = None):
        super().__init__(message)
        self.hints = hints


def handle_tool_error(func):
    """
    Decorator that wraps tool functions with standardized error handling.

    Catches ToolValidationError (with hints), ValueError (without hints),
    and Exception (unexpected errors). Returns structured error responses
    matching existing behavior.

    Works with both sync and async functions.
    """
    if inspect.iscoroutinefunction(func):
        @functools.wraps(func)
        async def async_wrapper(*args, **kwargs):
            try:
                return await func(*args, **kwargs)
            except ToolValidationError as e:
                logger.warning(f"Validation error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.VALIDATION_ERROR,
                    message=str(e),
                    hints=e.hints,
                )
            except ValueError as e:
                logger.warning(f"Validation error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.VALIDATION_ERROR,
                    message=str(e),
                )
            except Exception as e:
                logger.exception(f"Unexpected error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.INTERNAL_ERROR,
                    message=f"Unexpected error: {str(e)}",
                )
        return async_wrapper
    else:
        @functools.wraps(func)
        def sync_wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except ToolValidationError as e:
                logger.warning(f"Validation error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.VALIDATION_ERROR,
                    message=str(e),
                    hints=e.hints,
                )
            except ValueError as e:
                logger.warning(f"Validation error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.VALIDATION_ERROR,
                    message=str(e),
                )
            except Exception as e:
                logger.exception(f"Unexpected error in {func.__name__}: {e}")
                return create_error_response(
                    error_code=ErrorCode.INTERNAL_ERROR,
                    message=f"Unexpected error: {str(e)}",
                )
        return sync_wrapper


def slim_response(response: dict) -> dict:
    """
    Remove redundant raw XML when parsed data exists.

    When SLIM_RESPONSE=true:
    - Removes 'raw_content' (XML) only if 'ranked_data' exists as replacement

    When not set, returns response unchanged.

    Args:
        response: The response dictionary from a tool

    Returns:
        Response with raw XML removed (if parsed data available), or unchanged
    """
    if not os.getenv("SLIM_RESPONSE"):
        return response

    result = response.copy()

    # Remove raw_content only if ranked_data exists as replacement
    # Without this guard, failed ranking chains leave empty responses
    if "ranked_data" in result:
        result.pop("raw_content", None)

    return result
```

### Task 3: Create `_helpers.py` — `run_search` Helper

**Files:**
- Modify: `src/lexlink/_helpers.py`

This is the most critical helper. It must handle **three distinct ranking pipelines** observed in existing tools:

**Pipeline A — Law-list:** `eflaw_search`, `law_search`, `elaw_search`, `lnkLs_search`
- Extracts via `extract_law_list()`, updates via `update_law_list()`
- Over-fetches via `numOfRows`
- Optionally calls `_cache_law()` post-ranking

**Pipeline B — Items-list:** `prec_search`, `detc_search`, `expc_search`, `decc_search`
- Extracts via `extract_items_list(parsed_data, category)`
- Updates via `update_items_list(parsed_data, items, category)`
- Over-fetches via `display`

**Pipeline C — Inline-key:** `admrul_search`
- Accesses `parsed_data.get(key, [])` directly
- Updates `parsed_data[key] = ranked_items`
- Over-fetches via `numOfRows`

**Pipeline D — Parse-only (no ranking):** `lnkLsOrdJo_search`, `lnkDep_search`, `aiSearch`, `aiRltLs_search`
- Just parse XML and set `ranked_data`, no ranking applied

- [ ] **Step 1: Add `run_search` to `_helpers.py`**

Append to `_helpers.py`:

```python
from .client import LawAPIClient
from .params import map_params_to_upstream, resolve_oc
from .parser import parse_xml_response, extract_law_list, update_law_list, extract_items_list, update_items_list
from .ranking import rank_search_results, should_apply_ranking


def run_search(
    *,
    get_client,
    target: str,
    query: str,
    snake_params: dict,
    response_type: str = "XML",
    display: int = 20,
    # Ranking configuration
    ranking_field: Optional[str] = None,
    list_type: Optional[str] = None,
    item_category: Optional[str] = None,
    over_fetch_key: str = "numOfRows",
    over_fetch: bool = True,
    # Post-ranking hook (e.g., _cache_law for law searches)
    post_rank_hook=None,
) -> dict:
    """
    Common search tool logic.

    Handles: OC resolution, param mapping, API call, XML parsing,
    relevance ranking (3 pipelines), trimming, slim_response.

    Args:
        get_client: Callable that returns a LawAPIClient instance
        target: API target (e.g., "eflaw", "prec", "admrul")
        query: Search query string
        snake_params: Pre-built parameter dict (must include "oc", "target", etc.)
        response_type: "XML" or "HTML"
        display: Original requested display count
        ranking_field: Field name for relevance ranking (e.g., "법령명한글", "판례명").
                       If None, ranking is skipped (parse-only mode).
        list_type: "law" for extract_law_list, "items" for extract_items_list,
                   or a direct key string (e.g., "admrul") for inline extraction.
                   If None, ranking is skipped.
        item_category: Category for extract_items_list (e.g., "prec", "detc").
                       Required when list_type="items".
        over_fetch_key: Upstream param key for over-fetching ("numOfRows" or "display")
        post_rank_hook: Optional callback(ranked_items) called after ranking.
                        Used for _cache_law in law searches.

    Returns:
        Tool response dict with status, raw_content, and optional ranked_data
    """
    # Map to upstream format
    upstream_params = map_params_to_upstream(snake_params)

    # Determine if ranking should be applied
    original_display = display
    ranking_enabled = (
        response_type == "XML"
        and ranking_field is not None
        and list_type is not None
        and should_apply_ranking(query)
    )

    if ranking_enabled and over_fetch and original_display < 100:
        upstream_params[over_fetch_key] = "100"
        logger.debug(f"Ranking enabled: fetching 100 results instead of {original_display}")

    # Execute request
    client = get_client()
    response = client.get("/DRF/lawSearch.do", upstream_params, response_type)

    # Parse XML and optionally rank
    if response.get("status") == "ok" and response_type == "XML":
        raw_content = response.get("raw_content", "")
        if raw_content:
            parsed_data = parse_xml_response(raw_content)
            if parsed_data:
                if ranking_enabled:
                    # Extract items based on list_type
                    if list_type == "law":
                        items = extract_law_list(parsed_data)
                    elif list_type == "items":
                        items = extract_items_list(parsed_data, item_category)
                    else:
                        # Inline key access (e.g., "admrul")
                        items = parsed_data.get(list_type, [])
                        if not isinstance(items, list):
                            items = [items] if items else []

                    if items:
                        ranked_items = rank_search_results(items, query, ranking_field)

                        # Trim to original requested display
                        if len(ranked_items) > original_display:
                            ranked_items = ranked_items[:original_display]
                            logger.debug(f"Trimmed results from {len(items)} to {original_display}")

                        # Update parsed data
                        if list_type == "law":
                            parsed_data = update_law_list(parsed_data, ranked_items)
                            parsed_data["numOfRows"] = str(original_display)
                        elif list_type == "items":
                            parsed_data = update_items_list(parsed_data, ranked_items, item_category)
                            parsed_data["display"] = str(original_display)
                        else:
                            # Inline key
                            parsed_data[list_type] = ranked_items
                            parsed_data["numOfRows"] = str(original_display)

                        response["ranked_data"] = parsed_data

                        # Post-ranking hook (e.g., _cache_law)
                        if post_rank_hook:
                            post_rank_hook(ranked_items)
                else:
                    # Parse-only mode (no ranking)
                    response["ranked_data"] = parsed_data

    return slim_response(response)


def run_service(
    *,
    get_client,
    target: str,
    snake_params: dict,
    response_type: str = "XML",
) -> dict:
    """
    Common service tool logic.

    Handles: param mapping, API call, return response.

    Args:
        get_client: Callable that returns a LawAPIClient instance
        target: API target (e.g., "eflaw", "prec", "admrul")
        snake_params: Pre-built parameter dict (must include "oc", "target", etc.)
        response_type: "XML" or "HTML"

    Returns:
        Tool response dict
    """
    upstream_params = map_params_to_upstream(snake_params)

    client = get_client()
    return client.get("/DRF/lawService.do", upstream_params, response_type)
```

```python
    ) -> dict:
        """
        Search ministry legal interpretations (중앙부처 1차 법령해석 목록 조회).

        Search legal interpretations issued directly by government ministries.
        Unlike expc_search (법령해석례 from 법제처), these are first-instance
        interpretations from individual ministries.

        Args:
            ministry: Ministry name in Korean (e.g., "고용노동부", "국토교통부").
                See tool description for full list of 39 ministries.
            query: Search keyword (default "*")
            display: Number of results per page (max 100, default 20)
            page: Page number (1-based, default 1)
            oc: Optional OC override
            type: Response format - "HTML" or "XML" (default "XML")
            search: 1=사건명 (case name), 2=본문검색 (full text)
            sort: Sort order
            gana: Dictionary search

        Returns:
            Ministry interpretation list or error

        Examples:
            Search 고용노동부 interpretations:
            >>> cgm_expc_search(ministry="고용노동부", query="근로시간")
        """
        code = MINISTRY_CODES.get(ministry)
        if not code:
            valid = ", ".join(MINISTRY_CODES.keys())
            raise ValueError(f"Invalid ministry: '{ministry}'. Valid values: {valid}")

        target = f"cgmExpc{code}"
        resolved_oc = resolve_oc(override_oc=oc)
        snake_params = {
            "oc": resolved_oc, "target": target, "type": type,
            "query": query, "display": display, "page": page,
        }
        if search: snake_params["search"] = search
        if sort: snake_params["sort"] = sort
        if gana: snake_params["gana"] = gana

        return run_search(
            get_client=_get_client, target=target, query=query,
            snake_params=snake_params, response_type=type, display=display,
            ranking_field="사건명", list_type="items", item_category=target,
            over_fetch_key="display",
        )
```
